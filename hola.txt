 test01ListCartFailsIfCartIdDoesNotExist
	| temp1 |
	temp1 _ testObjectsFactory createRESTAPI.
	self
		should: [ temp1 listCartWithID: 'invalid_id' ]
		raise: Error - MessageNotUnderstood
		withExceptionDo: [ :arg2 |
			self
				assert: arg2 messageText
				equals: TusLibrosRESTAPI cartNotFoundErrorDescription ].
